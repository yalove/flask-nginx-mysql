#mysql 
MYSQL_USER=admin

MYSQL_PASS=**Random

`docker run -d --env MYSQL_USER=admin --env MYSQL_PASS=123 <images>`

set 
`MYSQL_USER= os.environ.get('MYSQL_USER')`

`MYSQL_PASS= os.environ.get('MYSQL_PASS')`
